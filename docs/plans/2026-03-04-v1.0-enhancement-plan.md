```rust
// crates/octo-engine/src/auth/api_key.rs

use uuid::Uuid;

pub struct ApiKey {
    pub id: String,
    pub key: String,           // 存储 hash
    pub user_id: String,
    pub role: UserRole,
    pub created_at: DateTime<Utc>,
    pub expires_at: Option<DateTime<Utc>>,
}

impl ApiKey {
    pub fn generate(user_id: &str, role: UserRole) -> Self {
        let key = Uuid::new_v4().to_string();
        let hash = Self::hash_key(&key);
        Self {
            id: Uuid::new_v4().to_string(),
            key: hash,
            user_id: user_id.to_string(),
            role,
            created_at: Utc::now(),
            expires_at: None,
        }
    }

    pub fn verify(&self, key: &str) -> bool {
        self.hash_key(key) == self.key
    }
}
```

**Step 1: 定义 Metering**

```rust
// crates/octo-engine/src/metering/mod.rs

use std::sync::atomic::{AtomicU64, Ordering};

pub struct Metering {
    pub input_tokens: AtomicU64,
    pub output_tokens: AtomicU64,
    pub requests: AtomicU64,
    pub errors: AtomicU64,
    pub duration_ms: AtomicU64,
}

impl Metering {
    pub fn record_request(&self, input: usize, output: usize, duration_ms: u64) {
        self.input_tokens.fetch_add(input as u64, Ordering::Relaxed);
        self.output_tokens.fetch_add(output as u64, Ordering::Relaxed);
        self.requests.fetch_add(1, Ordering::Relaxed);
        self.duration_ms.fetch_add(duration_ms, Ordering::Relaxed);
    }

    pub fn snapshot(&self) -> MeteringSnapshot {
        MeteringSnapshot {
            input_tokens: self.input_tokens.load(Ordering::Relaxed),
            output_tokens: self.output_tokens.load(Ordering::Relaxed),
            requests: self.requests.load(Ordering::Relaxed),
            errors: self.errors.load(Ordering::Relaxed),
            duration_ms: self.duration_ms.load(Ordering::Relaxed),
        }
    }
}
```

**Step 2: Provider 集成**

```rust
// 在 Provider trait 中添加 metering 回调

pub trait Provider: Send + Sync {
    async fn complete(&self, request: Request) -> Result<Response> {
        let start = Instant::now();
        let result = self.complete_raw(request).await;
        let duration = start.elapsed();

        if let Some(metering) = &self.metering {
            let input = request.input_tokens();
            let output = result.as_ref().map(|r| r.output_tokens()).unwrap_or(0);
            metering.record_request(input, output, duration.as_millis() as u64);
        }

        result
    }
}
```
